apiVersion: apps/v1
kind: Deployment
metadata:
  name: trust-graph-api
spec:
  replicas: 2
  selector:
    matchLabels:
      app: trust-graph-api
  template:
    metadata:
      labels:
        app: trust-graph-api
    spec:
      containers:
        - name: api
          image: yourregistry/trust-graph:latest
          imagePullPolicy: IfNotPresent
          command: ["uvicorn", "app.main:app", "--host", "0.0.0.0", "--port", "8000"]
          env:
            - name: PORT
              value: "8000"
            - name: NEO4J_URI
              valueFrom:
                secretKeyRef:
                  name: neo4j-secret
                  key: uri
            - name: NEO4J_USER
              valueFrom:
                secretKeyRef:
                  name: neo4j-secret
                  key: user
            - name: NEO4J_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: neo4j-secret
                  key: password
            - name: REDIS_URL
              value: "redis://trust-graph-redis:6379/0"
          ports:
            - containerPort: 8000
---
apiVersion: v1
kind: Service
metadata:
  name: trust-graph-api
spec:
  selector:
    app: trust-graph-api
  ports:
    - protocol: TCP
      port: 80
      targetPort: 8000
  type: ClusterIP
